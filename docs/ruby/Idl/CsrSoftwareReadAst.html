def type(symtab, archdef)
  if csr_known?(symtab, archdef)
    Type.new(:bits, width: archdef.csr(csr.csr_name(symtab, archdef)).length)
  else
    Type.new(:bits, width: archdef.config_params["XLEN"])
  end
end
